Capstone Project – Task Manager App
Objectives
By the end of this project, students will be able to:
1. Combine multiple Android concepts (UI, fragments, storage, navigation, and
preferences) into one app.
2. Build a Task Manager App that allows users to add, view, edit, and delete tasks.
3. Implement SQLite database for data persistence.
4. Use SharedPreferences for app settings.
5. Apply fragments and navigation for better user experience.
What the App Will Do
The Task Manager App will allow users to:
•
•
•
•
•
•
Add a new task (title + description).
Save tasks permanently in SQLite.
View all tasks in a list.
Edit or delete tasks.
Navigate between different screens using fragments.
Store simple settings (e.g., dark mode toggle) using SharedPreferences.
Step-by-Step Instructions

1. Open Android Studio → New Project → Select Empty Activity → Name it
TaskManagerApp.
2. Language: Kotlin.
3. Minimum SDK: API 24 (Android 7.0) or higher.
4. Finish and wait for Gradle sync.

Step 4: Set Up SQLite Database
1. Create TaskDatabaseHelper.kt.
2. Extend SQLiteOpenHelper.
class TaskDatabaseHelper(context: Context) :
SQLiteOpenHelper(context, "tasks.db", null, 1) {
override fun onCreate(db: SQLiteDatabase?) {
db?.execSQL("CREATE TABLE tasks (id INTEGER PRIMARY KEY AUTOINCREMENT, title TEXT,
description TEXT)"
)
}
override fun onUpgrade(db: SQLiteDatabase?, oldVersion: Int, newVersion: Int) {
db?.execSQL("DROP TABLE IF EXISTS tasks")
onCreate(db)
}
fun insertTask(task: Task) {
val db = writableDatabase
val values = ContentValues().apply {
put("title", task.title)
put("description", task.description)
}
db.insert("tasks", null, values)
db.close()
}
fun getAllTasks(): List<Task> {
val taskList = mutableListOf<Task>()
val db = readableDatabase
val cursor = db.rawQuery("SELECT * FROM tasks", null)
if (cursor.moveToFirst()) {
do {
taskList.add(
Task(
id = cursor.getInt(0),
title = cursor.getString(1),
description = cursor.getString(2)
)
)
} while (cursor.moveToNext())
}
cursor.close()
db.close()
return taskList
}
fun deleteTask(id: Int) {
val db = writableDatabase
db.delete("tasks", "id=?", arrayOf(id.toString()))
db.close()}
}
Step 5: Build the Task List Fragment
•
•
Use a ListView (simple enough for undergraduates; RecyclerView could be taught
later).
Display tasks from database.
class TaskListFragment : Fragment() {
private lateinit var db: TaskDatabaseHelper
private lateinit var listView: ListView
private lateinit var adapter: ArrayAdapter<String>
override fun onCreateView(
inflater: LayoutInflater, container: ViewGroup?, savedInstanceState: Bundle?
): View? {
val view = inflater.inflate(R.layout.fragment_task_list, container, false)
db = TaskDatabaseHelper(requireContext())
listView = view.findViewById(R.id.task_list)
loadTasks()
val fab: FloatingActionButton = view.findViewById(R.id.fab_add)
fab.setOnClickListener {
parentFragmentManager.beginTransaction()
.replace(R.id.fragment_container, AddTaskFragment())
.addToBackStack(null)
.commit()
}
}
return view
private fun loadTasks() {
val tasks = db.getAllTasks()
val taskTitles = tasks.map { it.title }
adapter = ArrayAdapter(requireContext(), android.R.layout.simple_list_item_1,
taskTitles)
listView.adapter = adapter
}
}Step 6: Build the Add Task Fragment

class AddTaskFragment : Fragment() {
private lateinit var db: TaskDatabaseHelper
override fun onCreateView(
inflater: LayoutInflater, container: ViewGroup?, savedInstanceState: Bundle?
): View? {
val view = inflater.inflate(R.layout.fragment_add_task, container, false)
db = TaskDatabaseHelper(requireContext())
val titleInput: EditText = view.findViewById(R.id.task_title)
val descInput: EditText = view.findViewById(R.id.task_desc)
val saveBtn: Button = view.findViewById(R.id.btn_save)
saveBtn.setOnClickListener {
val title = titleInput.text.toString()
val desc = descInput.text.toString()
if (title.isNotEmpty()) {
db.insertTask(Task(title = title, description = desc))
Toast.makeText(requireContext(), "Task Saved", Toast.LENGTH_SHORT).show()
parentFragmentManager.popBackStack()
} else {
Toast.makeText(requireContext(), "Title is required",
Toast.LENGTH_SHORT).show()
}
}
}
}
return view
